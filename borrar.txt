lass LikesController < ApplicationController

  def create
    @like = current_user.likes.new(like_params)

    was_saved = @like.save

    respond_to do |format|
      format.html do
        if !was_saved
          flash[:notice] = @like.errors.full_messages.to_sentence
        end
        redirect_to @like.micropost
      end
      format.turbo_stream
    end
  end

  def destroy
    @like = current_user.likes.find(params[:id])
    post = @like.micropost
    @like.destroy
    redirect_to post
  end

  private

  def like_params
    params.require(:like).permit(:micropost_id)
  end
end
